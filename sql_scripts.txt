--последовательности и триггеры
-- table QuestionsTopic
create sequence seq1
start with 1
increment by 1
maxvalue 999999;

CREATE or REPLACE FUNCTION qTopics() RETURNS trigger AS $qTopics$
	begin
			INSERT INTO QuestionsTopic select SEQ1.nextval into NEW.id from dual;
	end;
$qTopics$ LANGUAGE plpgsql;
